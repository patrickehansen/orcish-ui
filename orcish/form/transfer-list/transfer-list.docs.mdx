import { ThemeProvider} from 'styled-components';

A component that does something special and renders text in a div.

### Component usage
```js
<ThemeProvider theme={defaultTheme} >
  <TransferList 
    left={{
      title: 'Left Items',
      entries: []
    }} 
    right={{
      title: 'Right Items',
      entries: []
    }} 
  />
</ThemeProvider>
```
